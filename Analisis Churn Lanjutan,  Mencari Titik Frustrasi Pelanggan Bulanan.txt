SELECT 
    InternetService AS Jenis_Internet,
    COUNT(customerID) AS Total_Pelanggan,
    SUM(CASE WHEN Churn = 'Yes' THEN 1 ELSE 0 END) AS Pelanggan_Kabur,
    ROUND(SUM(CASE WHEN Churn = 'Yes' THEN 1.0 ELSE 0.0 END) / COUNT(customerID) * 100, 2) AS Persentase_Churn
FROM 
    data_telco
WHERE 
    Contract = 'Month-to-month'
GROUP BY 
    InternetService
ORDER BY 
    Persentase_Churn DESC;
